MySQL表操作

修改列名

alter table 表名 change colume 原列名 新列名 列类型

修改列类型

alter table 表名 modify colume 列名 新类型

删除列

alter table 表名 drop colume 列名

修改表名

alter table 表名 rename 【to 】 新表名

添加列

alter table 表名 add colume 列名 列类型

删除表

drop table [if exists] 表名

复制表

只复制表的结构

create table 新表名 like 旧表名

复制表的结构和数据

create table 新表名

select * from 旧表名 【where 筛选条件】

如果只要部分表的字段

create table 新表名

select id,name

from 旧表名

where 0          只需要让条件不满足，这样就差不到数据，只显示字段了！
